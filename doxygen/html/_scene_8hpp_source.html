48 inline double Scene::getDoppler(Ray3D &test_ray, Vector3D &hitNormal, Point3D &hitPoint, double RPM) const
49 {
51  double frequency = test_ray.frequency;
52 
53  //form vect in direction of rotation
54  Vector3D hit_point_radius_perp(-hitPoint.y(), hitPoint.x(), 0);
55 
56  //find radial distance
57  double hit_point_radius_dist = std::sqrt((hitPoint.x()*hitPoint.x()) + (hitPoint.y()*hitPoint.y()));
58 
59  //determine reflection make sure is a vector
60  Vector3D reflection = -test_ray.direction + (2 * hitNormal) * (test_ray.direction * hitNormal);
61 
62  double omega_r = (2*M_PI/60)*RPM;
63  double wave_length = SPEED_OF_LIGHT/frequency;
64 
65  frequency = frequency + 2*(((hit_point_radius_dist*omega_r)/wave_length)
66  *(hit_point_radius_perp.normalized()*reflection.normalized()));
67 
68  return frequency;
69 }
